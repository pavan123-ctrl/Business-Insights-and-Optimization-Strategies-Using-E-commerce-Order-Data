# Load and prepare data
df_orders = pd.read_csv('df_Orders.csv')
df_payments = pd.read_csv('df_Payments.csv')
df = pd.merge(df_orders[['order_id', 'order_purchase_timestamp']], df_payments[['order_id', 'payment_value']], on='order_id')
df['order_purchase_timestamp'] = pd.to_datetime(df['order_purchase_timestamp'])
df['Month'] = df['order_purchase_timestamp'].dt.to_period('M').dt.to_timestamp()
monthly_sales = df.groupby('Month')['payment_value'].sum().reset_index()
monthly_sales.set_index('Month', inplace=True)

# Import Exponential Smoothing
from statsmodels.tsa.holtwinters import SimpleExpSmoothing

# Fit Simple Exponential Smoothing model
exp_model = SimpleExpSmoothing(monthly_sales['payment_value'], initialization_method="heuristic").fit(
    smoothing_level=0.2,  # You can adjust smoothing (alpha) between 0 and 1
    optimized=False
)


# Plot corrected forecast
plt.figure(figsize=(10, 4))
monthly_sales['payment_value'].plot(label='Original', color='blue')
exp_model.fittedvalues.plot(label='Fitted Values', color='green')
plt.title('Simple Exponential Smoothing Forecast (Corrected)')
plt.xlabel('Date')
plt.ylabel('Sales')
plt.legend()
plt.grid(True)
plt.tight_layout()
plt.show()
